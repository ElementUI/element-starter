/**
 * @description 主题首页 接收 get 参数
 * @method fetch
 * @param {obj} params - 请求的参数.
 * @param {Number} page - 页数
 * @param {String} tab - 主题分类,目前有 ask share job good
 * @param {Number} limit - 每一页的主题数量
 * @paramm {String} drender - 当为 false 时，不渲染。默认为 true，渲染出现的所有 markdown 格式文本。
 */
export function fetch(method, url, params) {
    if (method === 'get') {
        return new Promise((resolve, reject) => {
            axios
                .get(url + '?' + params)
                .then((response) => {
                    resolve(response.data);
                }, (err) => {
                    reject(err);
                })
                .catch((error) => {
                    reject(error)
                })
        })
    } else if (method === 'post') {
        return new Promise((resolve, reject) => {
            axios
                .post(url, params)
                .then((response) => {
                    resolve(response.data);
                }, (err) => {
                    reject(err);
                })
                .catch((error) => {
                    reject(error)
                })
        })
    } else if (method === ':id') {
        return new Promise((resolve, reject) => {
            axios
                .get(url + '/' + params.id)
                .then((response) => {
                    resolve(response.data);
                }, (err) => {
                    reject(err);
                })
                .catch((error) => {
                    reject(error)
                })
        })
    } else {
        console.log('fetch error');
    }
}
